167       function endTag($xp, $tag)
168       {
169           switch ($tag) {
170               
171               case 'LOG4PHP:EVENTSET':
172               case LOG4PHP_LOGGER_DOM_CONFIGURATOR_XMLNS.':EVENTSET':
173                   $this->eventSetStarted = false;
174                   break;
175                   
176               case 'LOG4PHP:EVENT':
177               case LOG4PHP_LOGGER_DOM_CONFIGURATOR_XMLNS.':EVENT':
178                   if ($this->eventStarted) {
179                       $this->dispatch();
180                   }
181                   $this->eventStarted = false;
182                   break;
183               
184               case 'LOG4PHP:MESSAGE':
185               case LOG4PHP_LOGGER_DOM_CONFIGURATOR_XMLNS.':MESSAGE':
186                   $translatedMessage = strtr($this->event->message, $this->htmlEntitiesTT);
187                   $this->event->message = $translatedMessage;
188                   $this->messageStarted = false;
189                   break;
190               
191               case 'LOG4PHP:NDC':
192               case LOG4PHP_LOGGER_DOM_CONFIGURATOR_XMLNS.':NDC':
193                   $translatedNdc = strtr($this->event->ndc, $this->htmlEntitiesTT);            
194                   $this->event->ndc = $translatedNdc;            
195                   $this->ndcStarted = false;
196                   break;
197               
198           }
